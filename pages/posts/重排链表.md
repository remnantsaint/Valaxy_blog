![](https://cdn.jsdelivr.net/gh/remnantsaint/hexoImage@main/20230403182251.png)
题解：
##### 方法一：寻找链表中点+反转链表+合并链表
注意到结果正是把链表中点的后半反转后（不包括中点，且偶数结点中中点是中间两个结点的后者），邻一个插入到前半链表，空间复杂度为$O(1)$
代码如下：
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 */
class Solution
{
public:
    ListNode *middleNode(ListNode *head) // 找链表中点
    {
        ListNode *first = head, *second = head; // 快慢指针
        while (first != NULL && first->next != NULL)
        {
            first = first->next->next;
            second = second->next;
        }
        return second;
    }

    ListNode *reverseList(ListNode *head) // 迭代法将中点后反转链表
    {
        ListNode *prev = nullptr;
        ListNode *curr = head;
        while (curr != nullptr)
        {
            ListNode *nextTemp = curr->next;
            curr->next = prev;
            prev = curr;
            curr = nextTemp;
        }
        return prev;
    }

    void mergeList(ListNode *l1, ListNode *l2)
    {
        ListNode *l1_tmp;
        ListNode *l2_tmp;
        while (l1 != NULL && l2 != NULL)
        {
            l1_tmp = l1->next; // 因为要改变结点的next
            l2_tmp = l2->next; // 所以先用tmp结点存当前结点的next结点

            l1->next = l2;
            l1 = l1_tmp;

            l2->next = l1;
            l2 = l2_tmp;
        }
    }

    void reorderList(ListNode *head)
    {
        if (head == NULL)
        {
            return;
        }
        ListNode *mid = middleNode(head); // 找中点或偶数中点后一个结点
        ListNode *l1 = head;
        ListNode *l2 = mid->next; // 两段链表
        mid->next = NULL;         // 断链
        l2 = reverseList(l2);     // 反转链表
        mergeList(l1, l2);
    }
};
```
##### 方法二：线性表
由于链表不支持下标访问，所以无法随机访问链表的元素，可以用vector来当作线性表存链表结点
此方法空间复杂度$O(n)$，用线性表比较大
代码如下：
```c++
class Solution
{
public:
    void reorderList(ListNode *head)
    {
        if (head == NULL)
        {
            return;//因为是void型，直接用return结束
        }
        vector<ListNode *> vec;
        ListNode *node = head;
        while (node != nullptr)
        {
            vec.push_back(node);
            node = node->next;//线性表vec存链表结点
        }
        int i = 0, j = vec.size() - 1;
        while (i < j)
        {
            vec[i]->next = vec[j];
            i++;
            if (i == j)
            {
                break;
            }
            vec[j]->next = vec[i];
            j--;//用i++和j--可以找下一个变量和上一个变量
        }
        vec[i]->next = NULL;
    }
};
```
